Eksponat
  id int primary key
  tytul varchar
  typ varchar
  wysokosc int
  szerokosc int
  waga int
  idTworca (może być NULL) int references Artysta

Artysta
  id int primary key
  imie varchar
  nazwisko varchar
  rokUrodzenia int
  rokSmierci (może być NULL) int;

Galeria
  id int primary key

Sala
  numer int
  pojemnosc int
  idGaleria int references Galeria
  primary key (numer, idGaleria)

WystawaObjazdowa
  id int primary key
  miasto varchar
  dataRozpoczecia date
  dataZakonczenia date

Ekspozycja
  id int primary key
  idEksponat int references Eksponat
  idGaleria (może być NULL) int references Galeria
  nrSala (może być NULL) int references Sala
  idWystawaObjazdowa (może być NULL) int references Wystawa objazdowa
--reguły poprawności:
--jeśli idGaleria i nrSala not NULL, to idWystawaObjazdowa NULL
--jeśli idWystawaObjazdowa not NULL, to idGaleria i nrSala NULL


wyzwalacze przy dodawaniu ekspozycji:
żaden obraz nie powinien przebywać na wystawach objazdowych dłużej niż 30 dni rocznie;
muzeum powinno zawsze mieć w swoich galeriach lub w magazynie co najmniej jeden obraz każdego artysty.
